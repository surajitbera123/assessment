             if (arr[i]==arr[j])

            }
            
            
          }
          
          if(repeat){
            arr[i]=0;
          }
          
          
        }
        
      for(let k=0;k<arr.length;k++){

      }

  10. Min and Max

    11.     function arrayMin(arr) {
        var minimum=arr[0];
        var N=arr.length;
       for(var i=1;i<N;i++) { 
          if(arr[i]<minimum){
            minimum=arr[i];
           }
         }
         return minimum;
       }
       
       function arrayMax(arr) {
       var maximum=arr[0];
       var N=arr.length;
       for(var j=1;j<N;j++){
          if(arr[j]>maximum){
            maximum=arr[j];
         }
       }
       return maximum;
       }

  11. Birthday Game

    12.     function Birthday_Game(arr,D ,M) {
 
        let counter = 0;
         for(let i = 0; i < arr.length - 1; i++){
              let subArray = arr.slice(i, i+M);
              if (subArray.length === M) {
                 let sum = subArray.reduce((a,b) => a + b, 0);
                 if(sum === D){
                      counter += 1;
                  }
               }
         }
         return counter;
    }
