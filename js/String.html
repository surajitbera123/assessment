  9. Reverse the String

    10.     var Reverse_String = (str) => 
{
     var i=0;
    var count=0;
    while(str[i]!==undefined){
      i++;
      count++;
    }
    let newString="";
   for(var j=count-1;j>=0;j--){
    newString+=str[j];
   }
   return newString;
};
