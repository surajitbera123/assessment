var Minimal_Angle = (h, m) => 
{
   let dpm=360/60;
   let dph=360/12;
   let a=h*dph+m*0.5;
   let b=m*dpm;
   let angle=Math.abs(a-b);
   return Math.min(angle,360-angle)
};

{
      let sum=0;
      let p=N;
  for(let i=1;i<N;i++){
   if(N%i==0){
      sum=sum+i;
   }
   
  }

   return ("YES");
}else{
   return ("NO");
}

var Reverse_Number = (N) => 
{
  let s=0,r;
  while(N!==0){
    r=N%10;
    s=s*10+r;
    N=parseInt(N/10);
  }
  return s;
};
